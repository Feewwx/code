```c
#include <stdio.h>

int main() {
    enum Color { RED, GREEN, BLUE };
    int num = 10;
    enum Color c = RED;  // 定义枚举常量c,并初始化为RED
    printf("num = %d, c = %d\n", num, c);
                         // 枚举常量本质上是整型常量,RED的值为0,GREEN的值为1,BLUE的值为2
    RED = 100;           // err,枚举常量不可修改
    return 0;
}
```
