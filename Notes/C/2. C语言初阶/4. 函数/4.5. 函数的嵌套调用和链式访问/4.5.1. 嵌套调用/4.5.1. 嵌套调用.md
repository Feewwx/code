```c
#include <stdio.h>

void new_line() { 
    printf("hehe\n"); 
}

void three_line() {
    int i = 0;
    for (i = 0; i < 3; i++) {
      new_line();
    }
}

int main() {
    three_line();
    return 0;
}
```

#include <stdio.h>

int Add(int x, int y) {
    return x + y;
    int Sub(int x, int y) { return x - y; }
}  // 不是这样嵌套!

int main() {
    int a = 0;
    int b = 0;
    scanf("%d %d", &a, &b);
    printf("%d", Add(a, b));

}
