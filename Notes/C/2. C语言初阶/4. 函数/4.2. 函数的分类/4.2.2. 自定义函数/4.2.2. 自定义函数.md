int get_max(int x, int y) {
    return (x > y ? x : y);

void test() { 
    printf("hello\n");

>`交换两个整型变量`

```c
#include <stdio.h>

// 形式参数
void Swap(int x, int y) {
    int z = 0;
    z = x;
    x = y;
    y = z;
}  // 错误写法

// 当实参传递给形参的时候,形参是实参的一份临时拷贝
// 对形参的修改不会影响实参

void Swap(int *px, int *py) {
    int z = *px;
    *px = *py;
    *py = z;
}

int main() {
    int a = 0;
    int b = 0;
    // a和b为实参
    scanf("%d %d", &a, &b);
    // Swap(a, b);
    Swap(&a, &b);
    printf("%d %d", a, b);
    return 0;
}
```
