```c
#include <stdio.h>

int main() {
    int arr[] = {1, 2, 3, 4, 5, 6, 7, 8, 9, 0};
    int *p = arr;
    int sz = sizeof(arr) / sizeof(arr[0]);
    for (int i = 0; i < sz; i++) {
        printf("&arr[%d] = %p ======= p + %d = %p\n", i, &arr[i], i, p + i);
    }
    return 0;
}
```

```c
    int sz = sizeof(arr) / sizeof(arr[0]);
```
