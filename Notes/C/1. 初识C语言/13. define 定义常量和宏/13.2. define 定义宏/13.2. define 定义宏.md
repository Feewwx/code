```c
#include <stdio.h>

#define ADD(x, y)                     ((x) + (y))  // 宏定义
//      宏名(宏的参数,参数是无类型)   宏体
int Add(int x, int y) { return x + y; }            // 函数定义

int main() {
    int a = 10;
    int b = 20;
    int c = ADD(a, b);
    printf("c = %d\n", c);
    c = Add(a, b);
    printf("c = %d\n", c);
    return 0;
}
```
