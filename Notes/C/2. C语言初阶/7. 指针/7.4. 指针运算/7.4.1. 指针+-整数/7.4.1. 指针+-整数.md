```c
#include <stdio.h>
#include <string.h>

int my_strlen(char *str) {
    int count = 0;
    while (*str != '\0') {
        count++;
        str++;
    }
    return count;
}

int main() {
    int len = my_strlen("hello");
    printf("%d\n", len);
    return 0;
}
```
