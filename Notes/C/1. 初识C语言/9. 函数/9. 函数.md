###
>`求两个任意整数的和`
```c
#include <stdio.h>

// 完整写法
int Add(int x, int y) {
    int z = x + y;
    return z;
}

// 简化写法
int Add(int x, int y) {
    return x + y;
}

int main() {
    int n1 = 0;
    int n2 = 0;
    scanf("%d %d", &n1, &n2);  // 输入
    int sum = Add(n1, n2);     // 求和
    printf("%d\n", sum);
    return 0;
}
```
