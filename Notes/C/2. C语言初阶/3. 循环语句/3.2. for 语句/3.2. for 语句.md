```c
#include <stdio.h>

int main() {
    int i = 0;
    for (i = 1; i <= 10; i++) {
        if (i == 3) {
            continue;
        }
        if (i == 8) {
            break;
        }
        printf("%d ", i);
    }  // 1 2 4 5 6 7
    return 0;
}
```

> `forеҫӘзҺҜзҡ„еҲӨж–ӯйғЁеҲҶзңҒз•Ҙж„Ҹе‘ізқҖеҲӨж–ӯдјҡжҒ’жҲҗз«Ӣ`


```c
#include <stdio.h>

int main() {
    for (int i = 0; i < 10; i++) {
        for (int j = 0; j < 10; j++) {
            printf("hehe\n");
        }
    }  // еҫӘзҺҜ10*10=100ж¬Ў
    return 0;
}
```

```c
#include <stdio.h>

int main() {
    int x, y;
    for (x = 0, y = 0; x < 2 && y < 5; x++, y++) {
        printf("hehe\n");
    }  // еӨҡдёӘеҸҳйҮҸ
    return 0;
}
```
