> 字符指针中存放的是字符串首字符的地址

```c
#include <stdio.h>

int main() {
    const char *p1 = "abc";
    const char *p2 = "abc";

    char *p3 = "abc";
    char *p4 = "abc";
  
    char arr5[] = "abc";
    char arr6[] = "abc";
    char *p5 = arr5;
    char *p6 = arr6;
    
    if (p1 == p2)
        printf("ok");
    else  // ok
        printf("no");
    
    if (p3 == p4)
        printf("ok");
    else  // ok
        printf("no");
    
    if (p5 == p6)
        printf("ok");
    else  // no
        printf("no");
   
    return 0;
}
```
