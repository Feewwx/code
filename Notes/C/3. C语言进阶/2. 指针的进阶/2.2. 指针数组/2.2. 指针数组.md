####
- 存放指针的数组就是指针数组

```c
#include <stdio.h>

int main() {
    int a = 10;
    int b = 20;
    int c = 30;
    int *parr[] = {&a, &b, &c};  // 指针数组
    return 0;
}
```
####
```c
#include <stdio.h>

int main() {
    int arr1[4] = {1, 2, 3, 4};
    int arr2[4] = {2, 3, 4, 5};
    int arr3[4] = {3, 4, 5, 6};

    int *parr[3] = {arr1, arr2, arr3};  // 整型指针数组

    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 4; j++) {
            printf("%d ", parr[i][j]);  // parr[i][j] = *(*(parr + i) + j)
        }  // * 和 [] 都是解引用
        printf("\n");
    }
    return 0;
}
```
