```c
#include <stdio.h>

int main() {
    int a = 10;          // 向内存申请4个字节,储存10  
    &a;                  // 取地址操作符
    printf("%p\n", &a);  // 0x7fff7ee5a304,地址为第一个字节的地址
    int *p = &a;         // p就是指针变量
    // int 说明指针指向的是一个int类型的变量(a是int类型) *p 说明p是指针变量
    *p;                  // 解引用操作符,意思就是通过p中存放的地址,找到p所指向的对象,*p就是p指向的对象
    *p = 20;
    printf("%d\n", a);
    printf("%p\n", p);
    char ch = 'a';
    char *p1 = &ch;
    printf("%p\n", p1);
    return 0;
}
```
