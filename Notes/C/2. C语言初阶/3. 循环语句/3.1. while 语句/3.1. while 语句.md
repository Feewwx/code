> `打印1-10数字`

```c
#include <stdio.h>

int main() {
    int i = 1;
    while (i <= 10) {
        printf("%d\n", i);
        i++;
    }
    // 或者
    while (1) {
        if (i > 10) {
            break;  // 终止循环
        }
        continue;  // 跳过当前循环continue后面的代码,继续判断下一次循环
        printf("%d\n", i);
        i++;
    }
    return 0;
}
```

```c
#include <stdio.h>

int main() {
    int ch = 0;
    while ((ch = getchar()) != EOF) {  // EOF:end of file
        putchar(ch);
    }
    return 0;
}
```
